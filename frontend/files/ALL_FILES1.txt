  <script type="module">
    import { requireAuth } from "./js/auth.js";
    import { eventAPI } from "./js/api.js";

    requireAuth();

    const params = new URLSearchParams(window.location.search);
    const eventId = params.get('eventId');
    
    let currentEvent = null;
    let pendingAction = null;

    async function loadEvent() {
      try {
        const { event } = await eventAPI.getEvent(eventId);
        currentEvent = event;
        
        document.getElementById('eventTitle').textContent = event.eventName;
        renderSets(event.sets);
        loadStats();
        loadParticipants();
        
        // Auto-refresh stats every 5 seconds
        setInterval(() => {
          loadStats();
          loadParticipants();
        }, 5000);
      } catch (error) {
        console.error("Error loading event:", error);
        alert("Error loading event: " + error.message);
      }
    }

    function renderSets(sets) {
      const setsHTML = sets.map(set => `
        <div class="set-item ${set.isActive ? 'active' : ''}">
          <div>
            <h4>${set.setName}</h4>
            <p style="margin: 5px 0; color: var(--text-secondary);">
              <i class="fas fa-clock"></i> ${set.timeLimit} minutes
            </p>
            ${set.isActive ? '<span class="badge badge-available"><i class="fas fa-broadcast-tower"></i> LIVE</span>' : ''}
          </div>
          <button class="toggle-btn ${set.isActive ? 'deactivate' : 'activate'}" 
                  onclick="toggleSet('${set._id}', ${!set.isActive})">
            ${set.isActive ? 'Deactivate' : 'Activate'}
          </button>
        </div>
      `).join('');
      
      document.getElementById('setsContainer').innerHTML = setsHTML;
    }

    window.toggleSet = function(setId, enable) {
      pendingAction = { setId, enable };
      document.getElementById('passwordModal').classList.add('active');
    };

    window.closePasswordModal = function() {
      document.getElementById('passwordModal').classList.remove('active');
      document.getElementById('adminPassword').value = '';
      pendingAction = null;
    };

    window.confirmAction = async function() {
      const password = document.getElementById('adminPassword').value;
      
      if (!password) {
        alert('Please enter admin password');
        return;
      }

      try {
        await eventAPI.toggleSet({
          eventId,
          setId: pendingAction.setId,
          adminPassword: password,
          enable: pendingAction.enable
        });

        alert(pendingAction.enable ? 'Set activated!' : 'Set deactivated!');
        closePasswordModal();
        loadEvent(); // Refresh
      } catch (error) {
        alert('Error: ' + error.message);
      }
    };

    async function loadStats() {
      try {
        const { stats } = await eventAPI.getStats(eventId);
        
        document.getElementById('totalParticipants').textContent = stats.totalParticipants;
        document.getElementById('totalSubmissions').textContent = stats.totalSubmissions;
        document.getElementById('above80').textContent = stats.above80;
        document.getElementById('above50').textContent = stats.above50;
      } catch (error) {
        console.error("Error loading stats:", error);
      }
    }

    async function loadParticipants() {
      try {
        const { participants } = await eventAPI.getParticipants(eventId);
        
        if (participants.length === 0) {
          document.getElementById('participantsBody').innerHTML = 
            '<tr><td colspan="4" class="text-center">No participants yet</td></tr>';
          return;
        }

        const participantsHTML = participants.map(p => {
          const result = p.setResults[0];
          const score = result?.score ?? '-';
          const status = result?.completedAt ? '✅ Completed' : '⏳ In Progress';
          
          return `
            <tr>
              <td>${p.rollNo}</td>
              <td>${p.department}</td>
              <td>${score}</td>
              <td>${status}</td>
            </tr>
          `;
        }).join('');

        document.getElementById('participantsBody').innerHTML = participantsHTML;
      } catch (error) {
        console.error("Error loading participants:", error);
      }
    }

    window.deleteEventConfirm = function() {
      if (confirm('Are you sure you want to delete this event? This cannot be undone.')) {
        document.getElementById('passwordModal').classList.add('active');
        pendingAction = { delete: true };
      }
    };

    loadEvent();
  </script>

    import { requireAuth, setupLogoutButton } from "./js/auth.js";

    let currentLevel = "easy";

      "percentages": { icon: "fa-percent", description: "Calculate percentages and related problems" },

        if (data.topics.length === 0) {
          document.getElementById("topicsGrid").innerHTML = 
            '<p class="text-center">No topics available yet. Check back soon!</p>';
          return;
        }

        const topicsHTML = data.topics.map(topic => {

    import { requireAuth } from "./js/auth.js";
    import { eventAPI } from "./js/api.js";

    requireAuth();

    const params = new URLSearchParams(window.location.search);
    const eventId = params.get('eventId');
    const participantId = params.get('participantId');

    let currentSet = null;
    let questions = [];
    let answers = [];
    let timeRemaining = 0;
    let timerInterval = null;
    let pollInterval = null;

    async function init() {
      try {
        // Poll for active set
        pollInterval = setInterval(checkForActiveSet, 3000);
        await checkForActiveSet();
      } catch (error) {
        console.error("Initialization error:", error);
        alert("Error: " + error.message);
      }
    }

    async function checkForActiveSet() {
      try {
        const { activeSet } = await eventAPI.getActiveSet(eventId);
        
        if (activeSet) {
          clearInterval(pollInterval);
          await startQuiz(activeSet);
        }
      } catch (error) {
        console.error("Error checking active set:", error);
      }
    }

    window.checkAgain = function() {
      checkForActiveSet();
    };

    async function startQuiz(activeSet) {
      try {
        // Start the set
        const result = await eventAPI.startSet({
          participantId,
          setId: activeSet._id
        });

        currentSet = activeSet;
        questions = result.questions || [];
        answers = new Array(questions.length).fill(null);
        timeRemaining = result.timeLimit * 60;

        // Hide waiting, show quiz
        document.getElementById('waitingScreen').style.display = 'none';
        document.getElementById('quizInterface').style.display = 'block';

        // Update header
        document.getElementById('quizTitle').textContent = activeSet.setName;
        document.getElementById('quizInfo').textContent = 
          `${questions.length} Questions | ${activeSet.timeLimit} Minutes`;

        // Render questions
        renderQuestions();
        
        // Start timer
        startTimer();

        // Enter fullscreen
        openFullscreen(document.documentElement);
      } catch (error) {
        console.error("Error starting quiz:", error);
        alert("Error starting quiz: " + error.message);
      }
    }

    function renderQuestions() {
      const optionLabels = ['A', 'B', 'C', 'D'];
      
      const questionsHTML = questions.map((q, index) => `
        <div class="question-card">
          <div class="question-number">
            <i class="fas fa-question-circle"></i>
            Question ${index + 1} of ${questions.length}
          </div>
          <div class="question-text">${q.question}</div>
          <div class="options-container">
            ${q.options.map((option, optIndex) => `
              <div class="option" onclick="selectOption(${index}, '${option.replace(/'/g, "\\'")}')">
                <div class="option-label">${optionLabels[optIndex]}</div>
                <div>${option}</div>
              </div>
            `).join('')}
          </div>
        </div>
      `).join('');

      document.getElementById('questionsContainer').innerHTML = questionsHTML;
    }

    window.selectOption = function(questionIndex, option) {
      answers[questionIndex] = option;
      
      // Update UI
      const questionCards = document.querySelectorAll('.question-card');
      const optionElements = questionCards[questionIndex].querySelectorAll('.option');
      
      optionElements.forEach(el => {
        el.classList.remove('selected');
        if (el.textContent.includes(option)) {
          el.classList.add('selected');
        }
      });
    };

    function startTimer() {
      updateTimerDisplay();
      
      timerInterval = setInterval(() => {
        timeRemaining--;
        updateTimerDisplay();

        if (timeRemaining <= 60) {
          document.getElementById('timer').classList.add('warning');
        }

        if (timeRemaining <= 0) {
          clearInterval(timerInterval);
          alert("Time's up! Auto-submitting...");
          submitQuiz();
        }
      }, 1000);
    }

    function updateTimerDisplay() {
      const minutes = Math.floor(timeRemaining / 60);
      const seconds = timeRemaining % 60;
      document.getElementById('timeDisplay').textContent = 
        `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
    }

    window.submitQuiz = async function() {
      try {
        clearInterval(timerInterval);

        const unanswered = answers.filter(a => a === null).length;
        
        if (unanswered > 0) {
          if (!confirm(`You have ${unanswered} unanswered questions. Submit anyway?`)) {
            return;
          }
        }

        // Calculate score (simple client-side for now)
        const score = answers.filter(a => a !== null).length;

        await eventAPI.submitSet({
          participantId,
          setId: currentSet._id,
          score,
          answers
        });

        alert(`Quiz submitted! Your score: ${score}/${questions.length}`);
        window.location.href = 'challenge.html';
      } catch (error) {
        console.error("Error submitting quiz:", error);
        alert("Error submitting: " + error.message);
      }
    };

    function openFullscreen(element) {
      if (element.requestFullscreen) {
        element.requestFullscreen();
      } else if (element.mozRequestFullScreen) {
        element.mozRequestFullScreen();
      } else if (element.webkitRequestFullscreen) {
        element.webkitRequestFullscreen();
      } else if (element.msRequestFullscreen) {
        element.msRequestFullscreen();
      }
    }

    // Prevent leaving during quiz
    window.addEventListener('beforeunload', (e) => {
      if (document.getElementById('quizInterface').style.display === 'block') {
        e.preventDefault();
        e.returnValue = '';
      }
    });

    init();

    async function loadTopics() {
      try {
        const response = await fetch(`http://localhost:5000/api/practice/categories/${CATEGORY}/topics`);
        const data = await response.json();
        if (data.topics.length === 0) {
          document.getElementById("topicsGrid").innerHTML = '<p class="text-center">No topics available yet.</p>';
          return;
        }
        const topicsHTML = data.topics.map(topic => {
          const meta = topicMeta[topic] || { icon: "fa-book", description: "Practice questions" };
          return `<div class="topic-card" onclick="selectTopic('${topic}')">
            <h3><i class="fas ${meta.icon}"></i> ${formatTopicName(topic)}</h3>
            <p>${meta.description}</p>
            <div class="topic-stats">
              <span><i class="fas fa-layer-group"></i> Multiple sets</span>
              <span><i class="fas fa-question-circle"></i> 10 Qs/set</span>

              <h4>${statusIcon} Set ${set.setNumber} ${statusBadge}</h4>
              <p><i class="fas fa-clock"></i> ${set.timeLimit} minutes | <i class="fas fa-question-circle"></i> ${set.questionCount} questions</p>
            </div>
            <div>${actionBtn}</div>
          </div>`;
        }).join('');
        document.getElementById("setsContainer").innerHTML = setsHTML;

        console.error("Error loading sets:", error);

      }
